(talet är ett primtal) */ return 1; } int main(){ //test av funktion: printf("Ange ett tal att testa: "); int tal; scanf("%d", &tal); //if-sats som triggar om funktionen primtal returnerar 1 if(primtal(tal)){ printf("Talet är ett primtal!"); } else { printf("Talet är inte ett primtal!"); } return 0; }
